```yaml
name: Windows CI

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]

jobs:
  test-windows:
    runs-on: windows-latest
    strategy:
      matrix:
        python-version: ['3.9', '3.10', '3.11', '3.12']
        node-version: ['18.x', '20.x']
    
    steps:
      - uses: actions/checkout@v3
        with:
          # Important: Handle line endings for Windows
          core.safecrlf: false
      
      - name: Set up Python ${{ matrix.python-version }}
        uses: actions/setup-python@v4
        with:
          python-version: ${{ matrix.python-version }}
          cache: 'pip'
      
      - name: Set up Node.js ${{ matrix.node-version }}
        uses: actions/setup-node@v3
        with:
          node-version: ${{ matrix.node-version }}
          cache: 'npm'
      
      - name: Install Python dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt
          pip install -r requirements-dev.txt
      
      - name: Install Node dependencies
        run: npm ci
      
      - name: Run Python tests
        run: |
          python -m pytest tests/ -v --tb=short
        env:
          PYTHONPATH: ${{ github.workspace }}/src
      
      - name: Run Node tests
        if: matrix.node-version == '20.x'
        run: npm test
      
      - name: Run linting checks
        run: |
          pip install flake8 pylint
          flake8 src/ --count --select=E9,F63,F7,F82 --show-source --statistics
      
      - name: Verify cross-platform paths
        shell: pwsh
        run: |
          # Verify important paths exist
          if (-not (Test-Path "src")) { throw "src directory not found" }
          if (-not (Test-Path "tests")) { throw "tests directory not found" }
          Write-Host "Cross-platform paths verified"
      
      - name: Check line endings
        shell: pwsh
        run: |
          # Verify .gitattributes is configured
          if (Test-Path ".gitattributes") {
            Write-Host ".gitattributes found"
            Get-Content .gitattributes
          } else {
            Write-Host "Warning: .gitattributes not configured for line ending handling"
          }
      
      - name: Upload test results
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: windows-test-results-py${{ matrix.python-version }}-node${{ matrix.node-version }}
          path: |
            test-results.xml
            .coverage
          retention-days: 30

  build-windows:
    runs-on: windows-latest
    needs: test-windows
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'
      
      - name: Build distribution
        run: |
          pip install build
          python -m build
      
      - name: Verify build artifacts
        shell: pwsh
        run: |
          $artifacts = Get-ChildItem dist/ -Recurse
          if ($artifacts) {
            Write-Host "Build artifacts created:"
            $artifacts | ForEach-Object { Write-Host $_.FullName }
          } else {
            throw "No build artifacts found"
          }
```

```python
"""
Cross-platform utility functions for Windows/Linux/macOS compatibility.
"""
import os
import sys
from pathlib import Path
from typing import Union

def normalize_path(path: Union[str, Path]) -> Path:
    """
    Normalize path for current operating system.
    Handles both forward and backward slashes.
    """
    return Path(path).resolve()

def get_platform_specific_path(base_path: str, *components: str) -> Path:
    """
    Build path using os.path.join for proper OS handling.
    """
    full_path = os.path.join(base_path, *components)
    return Path(full_path)

def is_windows() -> bool:
    """Check if running on Windows."""
    return sys.platform.startswith('win')

def is_unix() -> bool:
    """Check if running on Unix-like system (Linux/macOS)."""
    return sys.platform.startswith(('linux', 'darwin'))

def get_line_ending() -> str:
    """Get appropriate line ending for current OS."""
    return '\r\n' if is_windows() else '\n'

def normalize_line_endings(text: str) -> str:
    """Normalize line endings to current OS standard."""
    return text.replace('\r\n', '\n').replace('\r', '\n')

def shell_escape(path: str) -> str:
    """Escape path for shell execution on current OS."""
    if is_windows():
        # Windows: quote if contains spaces
        if ' ' in path:
            return f'"{path}"'
    else:
        # Unix: use backslash escaping
        return path.replace(' ', '\\ ')
    return path

# Platform-specific imports
if is_windows():
    import msvcrt
    import ctypes
else:
    import termios
    import tty
```
