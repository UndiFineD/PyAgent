### File: tests/e2e/helpers/page-object.ts

/**
 * Base Page Object Model
 * Reusable page interactions
 */
import { Page, Locator } from '@playwright/test';

export class BasePage {
  constructor(protected page: Page) {}

  async goto(url: string) {
    await this.page.goto(url);
  }

  async waitForNavigation() {
    await this.page.waitForNavigation();
  }

  async click(selector: string) {
    await this.page.click(selector);
  }

  async fill(selector: string, text: string) {
    await this.page.fill(selector, text);
  }

  async getText(selector: string): Promise<string> {
    return await this.page.textContent(selector) || '';
  }

  async isVisible(selector: string): Promise<boolean> {
    return await this.page.isVisible(selector);
  }

  async waitForSelector(selector: string) {
    await this.page.waitForSelector(selector);
  }
}
```

/**
 * Authentication E2E Tests
 * Tests for login, logout, and authentication flows
 */
import { test, expect } from '@playwright/test';
import { BasePage } from './helpers/page-object';

test.describe('Authentication', () => {
  let page: BasePage;

  test.beforeEach(async ({ page: browserPage }) => {
    page = new BasePage(browserPage);
    await page.goto('/');
  });

  test('should login successfully', async ({ page: browserPage }) => {
    const loginPage = new BasePage(browserPage);
    
    await loginPage.click('[data-testid="login-button"]');
    await loginPage.fill('[data-testid="email-input"]', 'test@example.com');
    await loginPage.fill('[data-testid="password-input"]', 'password123');
    await loginPage.click('[data-testid="submit-button"]');

    await loginPage.waitForSelector('[data-testid="dashboard"]');
    expect(await loginPage.isVisible('[data-testid="dashboard"]')).toBeTruthy();
  });

  test('should show error on invalid credentials', async ({ page: browserPage }) => {
    const loginPage = new BasePage(browserPage);

    await loginPage.click('[data-testid="login-button"]');
    await loginPage.fill('[data-testid="email-input"]', 'invalid@example.com');
    await loginPage.fill('[data-testid="password-input"]', 'wrong');
    await loginPage.click('[data-testid="submit-button"]');

    const errorMsg = await loginPage.getText('[data-testid="error-message"]');
    expect(errorMsg).toContain('Invalid credentials');
  });

  test('should logout successfully', async ({ page: browserPage }) => {
    // Login first
    const loginPage = new BasePage(browserPage);
    await loginPage.click('[data-testid="login-button"]');
    await loginPage.fill('[data-testid="email-input"]', 'test@example.com');
    await loginPage.fill('[data-testid="password-input"]', 'password123');
    await loginPage.click('[data-testid="submit-button"]');

    // Then logout
    await loginPage.click('[data-testid="logout-button"]');
    await loginPage.waitForSelector('[data-testid="login-button"]');
    expect(await loginPage.isVisible('[data-testid="login-button"]')).toBeTruthy();
  });
});
```

### File: .github/workflows/e2e-tests.yml
```yaml
name: E2E Tests

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]
  schedule:
    - cron: '0 2 * * *'  # Daily at 2 AM UTC

jobs:
  test:
    timeout-minutes: 60
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v3
      
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'npm'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Install Playwright
        run: npx playwright install --with-deps
      
      - name: Run E2E tests
        run: npm run test:e2e
      
      - name: Upload test report
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: playwright-report
          path: playwright-report/
          retention-days: 30
      
      - name: Comment PR with results
        if: github.event_name == 'pull_request'
        run: |
          echo "E2E tests completed" >> $GITHUB_STEP_SUMMARY
```
