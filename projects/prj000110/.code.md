"""
Integration tests for API endpoints.
Tests complete request/response cycles with database.
"""
import pytest
from fastapi import status

class TestUserEndpoints:
    """User management API endpoint tests."""
    
    def test_create_user(self, api_client, db_session):
        """Test user creation endpoint."""
        response = api_client.post(
            "/api/users",
            json={"email": "newuser@example.com", "password": "secure123"}
        )
        assert response.status_code == status.HTTP_201_CREATED
        data = response.json()
        assert data["email"] == "newuser@example.com"
        assert "password" not in data
    
    def test_get_user(self, api_client, authenticated_client, user_factory):
        """Test user retrieval endpoint."""
        user = user_factory()
        response = authenticated_client.get(f"/api/users/{user.id}")
        assert response.status_code == status.HTTP_200_OK
        assert response.json()["email"] == user.email
    
    def test_list_users(self, authenticated_client, user_factory):
        """Test user listing endpoint."""
        user_factory()
        user_factory(email="user2@example.com")
        response = authenticated_client.get("/api/users")
        assert response.status_code == status.HTTP_200_OK
        assert len(response.json()) >= 2
    
    def test_update_user(self, authenticated_client, user_factory):
        """Test user update endpoint."""
        user = user_factory()
        response = authenticated_client.patch(
            f"/api/users/{user.id}",
            json={"email": "updated@example.com"}
        )
        assert response.status_code == status.HTTP_200_OK
        assert response.json()["email"] == "updated@example.com"
    
    def test_delete_user(self, authenticated_client, user_factory):
        """Test user deletion endpoint."""
        user = user_factory()
        response = authenticated_client.delete(f"/api/users/{user.id}")
        assert response.status_code == status.HTTP_204_NO_CONTENT
        
        # Verify deletion
        response = authenticated_client.get(f"/api/users/{user.id}")
        assert response.status_code == status.HTTP_404_NOT_FOUND

"""
Integration tests for database interactions.
Tests persistence, relationships, and transactions.
"""
import pytest
from sqlalchemy import exc

class TestDatabasePersistence:
    """Database persistence tests."""
    
    def test_user_persistence(self, db_session, user_factory):
        """Test user data persists in database."""
        user = user_factory(email="persist@example.com")
        user_id = user.id
        
        # Create new session
        from sqlalchemy.orm import sessionmaker
        Session = sessionmaker(bind=db_session.get_bind())
        new_session = Session()
        
        # Query from new session
        from src.models import User
        retrieved_user = new_session.query(User).filter_by(id=user_id).first()
        assert retrieved_user is not None
        assert retrieved_user.email == "persist@example.com"
        new_session.close()
    
    def test_relationship_loading(self, db_session, user_factory):
        """Test foreign key relationships."""
        user = user_factory()
        # Assuming user has posts relationship
        # posts = user.posts
        # assert len(posts) == 0
        assert user is not None

class TestTransactionHandling:
    """Transaction handling tests."""
    
    def test_rollback_on_error(self, db_session, user_factory):
        """Test rollback on constraint violation."""
        user1 = user_factory(email="unique@example.com")
        
        try:
            user2 = user_factory(email="unique@example.com")  # Duplicate
            db_session.commit()
            pytest.fail("Should raise IntegrityError")

            db_session.rollback()
