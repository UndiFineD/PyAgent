## Unified Observability Stack

### 1. Metrics Collection Integration

```python
# Integration wrapper referencing existing metrics infrastructure
from src.observability.metrics_engine import MetricsEngine
from prometheus_client import Counter, Histogram, Gauge

class UnifiedObservabilityStack:
    """Integrates metrics, tracing, logging, and alerting."""
    
    def __init__(self):
        self.metrics_engine = MetricsEngine()
        self.metrics = {
            'requests': Counter('http_requests_total', 'Total requests'),
            'latency': Histogram('http_request_duration_seconds', 'Request latency'),
            'errors': Counter('errors_total', 'Total errors')
        }
    
    def record_metric(self, name: str, value: float, labels: dict = None):
        """Record a metric using existing engine."""
        self.metrics_engine.record(name, value)
        if labels:
            self.metrics[name].labels(**labels).inc()
```

### 2. Distributed Tracing

```python
from opentelemetry import trace
from opentelemetry.exporter.jaeger.thrift import JaegerExporter

class TracingIntegration:
    """Integrates distributed tracing."""
    
    def __init__(self):
        self.tracer = trace.get_tracer(__name__)
        self.exporter = JaegerExporter()
    
    def start_span(self, name: str, attributes: dict = None):
        """Start a new trace span."""
        span = self.tracer.start_span(name)
        if attributes:
            for key, value in attributes.items():
                span.set_attribute(key, value)
        return span
```
