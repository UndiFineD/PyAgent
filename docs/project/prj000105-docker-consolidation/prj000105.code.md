name: Docker Configuration Check

on:
  push:
    branches: [ main, develop ]
    paths: [ 'deploy/docker-compose.yaml', '.env.template' ]
  pull_request:
    branches: [ main, develop ]
    paths: [ 'deploy/docker-compose.yaml', '.env.template' ]

jobs:
  docker-check:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v4
      
      - name: Check docker-compose syntax
        run: |
          docker-compose -f deploy/docker-compose.yaml config > /dev/null
        continue-on-error: true
      
      - name: Validate docker-compose file
        run: |
          if ! docker-compose -f deploy/docker-compose.yaml config > /dev/null; then
            echo "ERROR: docker-compose.yaml has syntax errors"
            exit 1
          fi
        continue-on-error: true
      
      - name: Check environment variables
        run: |
          grep -E "^\${[A-Z_]+}" deploy/docker-compose.yaml | sort -u > used_vars.txt
          grep -E "^[A-Z_]+=" .env.template | cut -d= -f1 | sort -u > defined_vars.txt
          echo "Used variables:"
          cat used_vars.txt
          echo "Defined variables:"
          cat defined_vars.txt
```

# Docker Compose Deployment Guide

## Quick Start

1. Copy environment template:
```bash
cp .env.template .env
```

2. Edit `.env` with your configuration

3. Start services:
```bash
docker-compose -f deploy/docker-compose.yaml up -d
```

4. Check health:
```bash
docker-compose -f deploy/docker-compose.yaml ps
```

## Services

### PyAgent Main Service
- Port: 8000
- Health check: /health endpoint
- Dependencies: Redis, PostgreSQL

### Redis Cache
- Port: 6379
- Health check: Redis ping
- Volume: redis-data

### PostgreSQL Database
- Port: 5432
- Health check: pg_isready
- Volume: postgres-data

## Scaling

To scale PyAgent service:
```bash
docker-compose -f deploy/docker-compose.yaml up -d --scale pyagent=3
```

## Monitoring

View logs:
```bash
docker-compose -f deploy/docker-compose.yaml logs -f pyagent
```

## Cleanup

Stop services:
```bash
docker-compose -f deploy/docker-compose.yaml down
```

Remove all data:
```bash
docker-compose -f deploy/docker-compose.yaml down -v
```
```
