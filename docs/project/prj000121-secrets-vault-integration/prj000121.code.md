### 1. Vault Client Wrapper (`src/security/vault_client.py`)

**New Module** - Lightweight wrapper around hvac client.

```python
"""Vault client for PyAgent secrets management."""

from typing import Any, Optional, Dict
import hvac
from src.core.exceptions import SecurityError

class VaultClient:
    """Lightweight Vault client wrapper with caching support."""
    
    def __init__(self, url: str, token: Optional[str] = None):
        """Initialize Vault client."""
        self.client = hvac.Client(url=url, token=token)
        self._cache: Dict[str, Any] = {}
    
    def read_secret(self, path: str, use_cache: bool = True) -> Dict[str, Any]:
        """Read secret from Vault."""
        if use_cache and path in self._cache:
            return self._cache[path]
        
        try:
            secret = self.client.secrets.kv.v2.read_secret_version(path=path)
            self._cache[path] = secret['data']['data']
            return secret['data']['data']
        except Exception as e:
            raise SecurityError(f"Failed to read secret: {str(e)}")
    
    def write_secret(self, path: str, data: Dict[str, Any]) -> None:
        """Write secret to Vault."""
        try:
            self.client.secrets.kv.v2.create_or_update_secret_version(
                path=path, secret_data=data
            )
            self._cache.pop(path, None)  # Invalidate cache
        except Exception as e:
            raise SecurityError(f"Failed to write secret: {str(e)}")
```

### 2. Vault Configuration (`src/security/vault_config.py`)

**New Module** - Configuration management for Vault.

```python
"""Vault configuration management."""

from dataclasses import dataclass
from typing import Optional

@dataclass
class VaultConfig:
    """Vault configuration."""
    address: str
    token: Optional[str] = None
    approle_id: Optional[str] = None
    approle_secret: Optional[str] = None
    namespace: Optional[str] = None
    tls_verify: bool = True
    
    @classmethod
    def from_env(cls) -> 'VaultConfig':
        """Load configuration from environment variables."""
        import os
        return cls(
            address=os.getenv('VAULT_ADDR', 'http://localhost:8200'),
            token=os.getenv('VAULT_TOKEN'),
            approle_id=os.getenv('VAULT_ROLE_ID'),
            approle_secret=os.getenv('VAULT_SECRET_ID'),
            namespace=os.getenv('VAULT_NAMESPACE'),
            tls_verify=os.getenv('VAULT_SKIP_VERIFY', 'false').lower() != 'true'
        )
```

### 3. Vault Authentication (`src/security/vault_auth.py`)

**New Module** - AppRole authentication.

```python
"""Vault AppRole authentication."""

from typing import Optional
import hvac

class VaultAppRole:
    """AppRole authentication handler."""
    
    def __init__(self, url: str, role_id: str, secret_id: str):
        """Initialize AppRole auth."""
        self.client = hvac.Client(url=url)
        self.role_id = role_id
        self.secret_id = secret_id
    
    def authenticate(self) -> str:
        """Authenticate and return token."""
        response = self.client.auth.approle.login(
            role_id=self.role_id,
            secret_id=self.secret_id
        )
        return response['auth']['client_token']
```

"""Integration tests for Vault client."""

import pytest
from unittest.mock import Mock, patch
from src.security.vault_client import VaultClient

    """Vault client tests."""

        """Test client init."""

        assert client is not None
    
    def test_read_secret_success(self):
        """Test successful secret read."""
        with patch('hvac.Client') as mock_hvac:
            mock_hvac.return_value.secrets.kv.v2.read_secret_version.return_value = {
                'data': {'data': {'key': 'value'}}
            }
            client = VaultClient(url="http://localhost:8200", token="test")
            result = client.read_secret("secret/test")
            assert result == {'key': 'value'}
    
    def test_read_secret_caching(self):
        """Test secret caching."""
        with patch('hvac.Client') as mock_hvac:
            client = VaultClient(url="http://localhost:8200", token="test")
            mock_hvac.return_value.secrets.kv.v2.read_secret_version.return_value = {
                'data': {'data': {'key': 'value'}}
            }
            client.read_secret("secret/test")
            client.read_secret("secret/test")
            # Should only call Vault once
            assert mock_hvac.return_value.secrets.kv.v2.read_secret_version.call_count == 1
