#!/usr/bin/env python3
"""Analyze and report on duplicate dependencies across requirements files."""

from __future__ import annotations

import json
import re
import sys
from collections import defaultdict
from dataclasses import asdict, dataclass
from pathlib import Path
from typing import Any


@dataclass
class Dependency:
    """Dependency information."""
    name: str
    version: str | None = None
    specifier: str | None = None
    file: str | None = None


class RequirementsAnalyzer:
    """Analyze requirements files for duplicates."""

    def __init__(self, root_dir: Path) -> None:
        """Initialize analyzer.
        
        Args:
            root_dir: Project root directory.
        """
        self.root_dir = Path(root_dir)
        self.dependencies: dict[str, list[Dependency]] = defaultdict(list)

    def find_requirements_files(self) -> list[Path]:
        """Find all requirements files in project.
        
        Returns:
            List of requirements file paths.
        """
        patterns = [
            "requirements.txt",
            "requirements-*.txt",
            "*/requirements.txt",
        ]
        
        files = []
        for pattern in patterns:
            files.extend(self.root_dir.glob(pattern))
        
        return sorted(set(files))

    def parse_requirement(self, line: str) -> Dependency | None:
        """Parse a requirement line.
        
        Args:
            line: Requirement line text.
            
        Returns:
            Dependency object or None if invalid.
        """
        line = line.strip()
        
        # Skip comments and empty lines
        if not line or line.startswith("#"):
            return None
        
        # Remove inline comments
        if "#" in line:
            line = line.split("#")[0].strip()
        
        # Parse package name and version specifier
        # Pattern: name[extras]specifier
        match = re.match(r"^([a-zA-Z0-9\-_.]+)", line)
        if not match:
            return None
        
        name = match.group(1)
        specifier = line[len(name):] if len(line) > len(name) else None
        
        return Dependency(name=name.lower(), specifier=specifier or None)

    def analyze_file(self, filepath: Path) -> None:
        """Analyze a requirements file.
        
        Args:
            filepath: Path to requirements file.
        """
        try:
            content = filepath.read_text()
            for line in content.split("\n"):
                dep = self.parse_requirement(line)
                if dep:
                    dep.file = str(filepath.relative_to(self.root_dir))
                    self.dependencies[dep.name].append(dep)
        except Exception as e:
            print(f"Error reading {filepath}: {e}", file=sys.stderr)

    def find_duplicates(self) -> dict[str, list[Dependency]]:
        """Find dependencies that appear in multiple files.
        
        Returns:
            Dictionary of duplicate dependencies.
        """
        return {
            name: deps
            for name, deps in self.dependencies.items()
            if len(set(d.file for d in deps)) > 1
        }

    def generate_report(self) -> dict[str, Any]:
        """Generate analysis report.
        
        Returns:
            Report dictionary.
        """
        duplicates = self.find_duplicates()
        
        return {
            "total_unique_packages": len(self.dependencies),
            "total_files": len(set(
                d.file for deps in self.dependencies.values() for d in deps
            )),
            "duplicate_packages": len(duplicates),
            "duplicates": {
                name: [asdict(d) for d in deps]
                for name, deps in sorted(duplicates.items())
            },
        }


def main() -> int:
    """Main entry point.
    
    Returns:
        Exit code (0 for success).
    """
    root_dir = Path.cwd()
    analyzer = RequirementsAnalyzer(root_dir)
    
    # Find and analyze files
    files = analyzer.find_requirements_files()
    print(f"Found {len(files)} requirements files", file=sys.stderr)
    
    for filepath in files:
        analyzer.analyze_file(filepath)
    
    # Generate and output report
    report = analyzer.generate_report()
    print(json.dumps(report, indent=2))
    
    # Exit with error if duplicates found
    if report["duplicate_packages"] > 0:
        print(f"\nERROR: Found {report['duplicate_packages']} duplicate packages", 
              file=sys.stderr)
        return 1
    
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

  dedup-check:
    runs-on: ubuntu-latest

      - uses: actions/checkout@v4
      
      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.12"
      
      - name: Run dependency deduplication check
        run: |
          python scripts/deps_dedup.py
        continue-on-error: true
      
      - name: Report findings
        if: always()
        run: |
          echo "Dependency analysis complete"

# Dependency Management Strategy

## Overview
This document describes the dependency management strategy for PyAgent and how to avoid duplicates across requirements files.

## Requirements File Hierarchy

1. **pyproject.toml**: Primary dependency source (main + dev)
2. **requirements.txt**: Generated or locked dependencies
3. **requirements-ci.txt**: CI-specific dependencies
4. **backend/requirements.txt**: Backend-specific dependencies

## Deduplication Rules

- Avoid specifying the same package in multiple files
- Use pyproject.toml [project.dependencies] as source of truth
- Generate lock files from pyproject.toml
- Use requirements-ci.txt only for CI-specific tools

## Checking for Duplicates

Run the dedup checker:
```bash
python scripts/deps_dedup.py
```

This will report:
- Total unique packages
- Number of files with duplicates
- Detailed duplicate list

## CI/CD Integration

The dedup check runs automatically on:
- Commits to main/develop
- Pull requests
- Changes to requirements files

Fix violations by consolidating duplicates in pyproject.toml.
