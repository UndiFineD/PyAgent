import click
from pathlib import Path

@click.group()
def cli():
    """PyAgent Developer Toolkit - Accelerate your workflow."""
    pass

@cli.command()
@click.option('--port', default=8000, help='Development server port')
def dev_server(port):
    """Start local development server."""
    # Reference existing FastAPI app
    from src.api.app import app
    import uvicorn
    uvicorn.run(app, host='0.0.0.0', port=port, reload=True)

@cli.command()
@click.option('--coverage', is_flag=True, help='Generate coverage report')
def test(coverage):
    """Run test suite."""
    # Reference existing test infrastructure
    import subprocess
    args = ['pytest', 'tests/']
    if coverage:
        args.extend(['--cov', '--cov-report=html'])
    subprocess.run(args)
